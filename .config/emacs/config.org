** Evil mode
#+begin_src elisp
;; Enable Evil
(use-package evil
  :elpaca t
  :init
  (setq evil-want-keybinding nil)
  (evil-mode))

;; Evil mode functionality for various other tools
(use-package evil-collection
  :elpaca t
  :after evil
  :config
  (setq evil-collection-mode-list '(dashboard dired))
  (evil-collection-init))
#+end_src
